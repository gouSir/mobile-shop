【项目概述】
  本项目是一个基于 Java Servlet + JSP 的手机在线购物商城系统。
  采用经典 MVC 分层架构：Model(实体类) → DAO(数据访问) → Servlet(控制层) → JSP(视图层)

  技术栈：
    - 后端：Servlet 4.0 + JSP 2.3 + JSTL 1.2
    - 数据库：MySQL 8.0 + JDBC（手动管理连接）
    - 前端：Bootstrap 5.3 + Bootstrap Icons
    - 构建：Maven + Tomcat 7 Maven Plugin
    - JDK：1.8

  项目结构：
    src/main/java/com/shop/
      ├── model/       实体类 (User, Product, CartItem, Order, OrderItem)
      ├── dao/         数据访问层 (UserDao, ProductDao, CartDao, OrderDao)
      ├── servlet/     控制层 Servlet (8个)
      ├── filter/      过滤器 (EncodingFilter)
      └── util/        工具类 (DBUtil)
    src/main/webapp/
      ├── WEB-INF/web.xml
      └── jsp/         视图页面 (含 admin/ 子目录)
    database/init.sql   数据库初始化脚本

  功能模块：
    用户模块     - 注册、登录、退出
    商品模块     - 商品列表、搜索、分类/品牌筛选、详情
    购物车模块   - 添加、修改数量、删除
    订单模块     - 创建订单、订单列表
    管理员模块   - 商品管理(CRUD)、订单管理(发货/完成/删除)

-- 创建数据库 mobile_shop（字符集 utf8mb4 支持中文和 Emoji）
CREATE DATABASE IF NOT EXISTS mobile_shop
  DEFAULT CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE mobile_shop;

-- ========== 1. 用户表 users ==========
-- 存储注册用户和管理员信息
-- role 字段区分身份：'user' 普通用户，'admin' 管理员
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,          -- 用户ID，自增主键
    username VARCHAR(50) NOT NULL UNIQUE,       -- 用户名，唯一
    password VARCHAR(50) NOT NULL,              -- 密码（明文存储，生产环境应加密）
    email VARCHAR(100),                         -- 电子邮箱
    phone VARCHAR(20),                          -- 手机号码
    address VARCHAR(200),                       -- 收货地址
    role VARCHAR(10) DEFAULT 'user',            -- 角色：user/admin
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP  -- 注册时间
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========== 2. 商品表 products ==========
-- 存储手机商品信息
CREATE TABLE products (
    id INT AUTO_INCREMENT PRIMARY KEY,          -- 商品ID
    name VARCHAR(100) NOT NULL,                 -- 商品名称
    brand VARCHAR(50),                          -- 品牌（Apple/Huawei/Xiaomi...）
    description TEXT,                           -- 商品描述
    price DECIMAL(10,2) NOT NULL,               -- 价格（精确到分）
    stock INT DEFAULT 0,                        -- 库存数量
    image VARCHAR(200),                         -- 图片文件名（存放于 images/ 目录）
    category VARCHAR(50),                       -- 分类（旗舰机/拍照手机/游戏手机/性价比）
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========== 3. 购物车表 cart ==========
-- 每个用户一个购物车，购物车项关联商品
-- 外键约束：删除用户/商品时级联删除对应购物车记录
CREATE TABLE cart (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,                       -- 用户ID（外键 → users.id）
    product_id INT NOT NULL,                    -- 商品ID（外键 → products.id）
    quantity INT DEFAULT 1,                     -- 购买数量
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========== 4. 订单表 orders ==========
-- 订单主表，记录每笔订单的汇总信息
CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,                       -- 下单用户（外键 → users.id）
    order_no VARCHAR(50) NOT NULL UNIQUE,       -- 订单编号（格式: MO+时间戳+用户ID）
    total_amount DECIMAL(10,2) NOT NULL,        -- 订单总金额
    status VARCHAR(20) DEFAULT '待发货',         -- 状态：待发货→已发货→已完成
    receiver_name VARCHAR(50),                  -- 收货人姓名
    receiver_phone VARCHAR(20),                 -- 收货人电话
    receiver_address VARCHAR(200),              -- 收货地址
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========== 5. 订单详情表 order_items ==========
-- 订单明细，记录每笔订单中每个商品的购买详情
CREATE TABLE order_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,                      -- 所属订单（外键 → orders.id）
    product_id INT NOT NULL,                    -- 商品ID（外键 → products.id）
    product_name VARCHAR(100),                  -- 商品名称（快照，防止商品被删后无法显示）
    product_price DECIMAL(10,2),                -- 商品单价（快照）
    quantity INT,                               -- 购买数量
    subtotal DECIMAL(10,2),                     -- 小计 = 单价 × 数量
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id)  -- 不级联删除，保留记录
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

package com.shop.model;

/**
 * 用户实体类 — 对应数据库 users 表
 * 字段：id, username, password, email, phone, address, role, createTime
 * role 字段区分用户身份：'user'(普通用户) / 'admin'(管理员)
 */
public class User {
    private int id;              // 用户ID（自增主键）
    private String username;     // 用户名（唯一）
    private String password;     // 密码
    private String email;        // 邮箱
    private String phone;        // 手机号
    private String address;      // 收货地址
    private String role;         // 角色: user 或 admin
    private String createTime;   // 注册时间

    // 两个构造器：无参构造（必须） + 全参构造（便捷）
    public User() {}

    public User(int id, String username, String password, String email,
                String phone, String address, String role, String createTime) {
        this.id = id;
        this.username = username;
        this.password = password;
        this.email = email;
        this.phone = phone;
        this.address = address;
        this.role = role;
        this.createTime = createTime;
    }

    // ★ Getter/Setter 方法 ★
    // JavaBean规范：属性通过 public getXxx()/setXxx() 暴露
    // JSP EL表达式（如 ${user.username}）通过 getUsername() 读取

    public int getId() { return id; }
    public void setId(int id) { this.id = id; }

    public String getUsername() { return username; }
    public void setUsername(String username) { this.username = username; }

    public String getPassword() { return password; }
    public void setPassword(String password) { this.password = password; }

    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }

    public String getPhone() { return phone; }
    public void setPhone(String phone) { this.phone = phone; }

    public String getAddress() { return address; }
    public void setAddress(String address) { this.address = address; }

    public String getRole() { return role; }
    public void setRole(String role) { this.role = role; }

    public String getCreateTime() { return createTime; }
    public void setCreateTime(String createTime) { this.createTime = createTime; }
}

package com.shop.model;

package com.shop.model;

/**
 * 购物车项实体 — 对应数据库 cart 表 + 关联查询的商品字段
 *
 * ★ 设计要点：
 *   1. 基础字段（id, userId, productId, quantity）来自 cart 表
 *   2. 关联属性（productName, productBrand, productPrice, productImage）
 *      通过 JOIN products 表查询获得，用于页面展示商品信息
 *   3. getSubtotal() 是计算属性，返回 单价×数量，不是数据库字段
 */
public class CartItem {
    private int id;               // 购物车记录ID
    private int userId;           // 用户ID
    private int productId;        // 商品ID
    private int quantity;         // 数量
    private String createTime;    // 加入时间

    // ★ 以下字段来自关联查询 products 表，用于页面展示
    private String productName;   // 商品名
    private String productBrand;  // 品牌
    private double productPrice;  // 单价
    private String productImage;  // 图片

    public CartItem() {}

    public int getId() { return id; }
    public void setId(int id) { this.id = id; }

    public int getUserId() { return userId; }
    public void setUserId(int userId) { this.userId = userId; }

    public int getProductId() { return productId; }
    public void setProductId(int productId) { this.productId = productId; }

    public int getQuantity() { return quantity; }
    public void setQuantity(int quantity) { this.quantity = quantity; }

    public String getCreateTime() { return createTime; }
    public void setCreateTime(String createTime) { this.createTime = createTime; }

    public String getProductName() { return productName; }
    public void setProductName(String productName) { this.productName = productName; }

    public String getProductBrand() { return productBrand; }
    public void setProductBrand(String productBrand) { this.productBrand = productBrand; }

    public double getProductPrice() { return productPrice; }
    public void setProductPrice(double productPrice) { this.productPrice = productPrice; }

    public String getProductImage() { return productImage; }
    public void setProductImage(String productImage) { this.productImage = productImage; }

    /**
     * ★ 计算属性：小计 = 单价 × 数量
     * 不用存储到数据库，每次调用时实时计算
     */
    public double getSubtotal() {
        return productPrice * quantity;
    }
}

package com.shop.model;

import java.util.ArrayList;
import java.util.List;

/**
 * 订单实体 — 对应数据库 orders 表
 *
 * ★ 一对多关系：一个订单(Order)包含多个订单明细(OrderItem)
 *   items 列表通过 OrderDao 加载
 *
 * ★ 订单编号规则：MO + 时间戳(yyyyMMddHHmmss) + 用户ID补零4位
 *   例如：MO202506271430520001
 */
public class Order {
    private int id;                 // 订单ID
    private int userId;             // 下单用户ID
    private String orderNo;         // 订单编号（唯一）

package com.shop.util;

import java.sql.*;

/**
 * 数据库连接工具类
 *
 * ★ 核心职责：
 *   1. 加载 MySQL JDBC 驱动
 *   2. 提供 getConnection() 获取数据库连接
 *   3. 提供 close() 方法安全关闭资源（防止连接泄漏）
 *
 * ★ 设计模式：
 *   - 使用 static 块加载驱动（只执行一次）
 *   - URL/USER/PASSWORD 硬编码（简单项目做法，生产环境应放配置文件）
 *
 * ★ 连接信息：
 *   数据库：localhost:3306/mobile_shop
 *   用户名：root
 *   密码：root
 *   参数：useSSL=false, serverTimezone=Asia/Shanghai, characterEncoding=utf-8
 */
public class DBUtil {

    // JDBC连接URL — 连接到本地MySQL的mobile_shop数据库
    // useSSL=false：关闭SSL（本地开发不需要加密连接）
    // serverTimezone=Asia/Shanghai：指定时区为东八区
    // characterEncoding=utf-8：指定通信编码
    private static final String URL =
        "jdbc:mysql://localhost:3306/mobile_shop?useSSL=false&serverTimezone=Asia/Shanghai&characterEncoding=utf-8";
    private static final String USER = "root";
    private static final String PASSWORD = "root";

    /**
     * ★ static 初始化块 — 类加载时执行一次
     * 加载 MySQL 8.x JDBC 驱动类 com.mysql.cj.jdbc.Driver
     * 注意：MySQL 5.x 驱动是 com.mysql.jdbc.Driver（已废弃）
     */
    static {
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
            throw new RuntimeException("MySQL驱动加载失败", e);
        }
    }

    /**
     * 获取数据库连接
     * ★ 每次调用都创建新连接，调用方用完必须 close()
     * 生产环境建议使用连接池（如 HikariCP、Druid）
     */
    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL, USER, PASSWORD);
    }

    /**
     * ★ 安全关闭资源（Connection + Statement + ResultSet）
     * 每个资源独立 try-catch，确保一个关闭失败不影响其他
     *
     * @param conn 数据库连接
     * @param stmt Statement 或 PreparedStatement
     * @param rs   结果集
     */
    public static void close(Connection conn, Statement stmt, ResultSet rs) {
        try {
            if (rs != null) rs.close();     // 先关 ResultSet
        } catch (SQLException e) {
            e.printStackTrace();
        }
        try {
            if (stmt != null) stmt.close(); // 再关 Statement
        } catch (SQLException e) {
            e.printStackTrace();
        }
        try {
            if (conn != null) conn.close(); // 最后关 Connection
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    /**
     * 重载：关闭 Connection 和 Statement（无 ResultSet 时使用）
     */
    public static void close(Connection conn, Statement stmt) {
        close(conn, stmt, null);
    }
}

package com.shop.filter;

import javax.servlet.*;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

/**
 * 编码过滤器 — 统一设置请求和响应的字符编码为 UTF-8
 *
 * ★ 为什么需要？
 *   如果没有这个过滤器，每个Servlet都要手动写：
 *     req.setCharacterEncoding("UTF-8");
 *     resp.setContentType("text/html;charset=UTF-8");
 *   使用过滤器后，一次配置，所有请求自动生效
 *
 * ★ web.xml 中配置 url-pattern 为 /* ，拦截所有请求
 * ★ doFilter 中的 chain.doFilter() 将请求传递给下一个过滤器或目标 Servlet
 */
public class EncodingFilter implements Filter {

    /** 过滤器初始化（无需额外配置） */
    @Override
    public void init(FilterConfig filterConfig) throws ServletException {
    }

    /**
     * ★ 核心方法：在每个请求到达 Servlet 之前执行
     *
     * 处理流程：
     *   1. 将通用 ServletRequest/Response 转为 HTTP 版本
     *   2. 设置请求编码为 UTF-8（让 req.getParameter() 正确读取中文）
     *   3. 设置响应内容类型为 text/html;charset=UTF-8（让浏览器正确渲染中文）
     *   4. 调用 chain.doFilter() 放行，继续后续处理
     */
    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {
        HttpServletRequest req = (HttpServletRequest) request;
        HttpServletResponse resp = (HttpServletResponse) response;

        req.setCharacterEncoding("UTF-8");
        resp.setContentType("text/html;charset=UTF-8");

        chain.doFilter(req, resp);  // ★ 必须调用，否则请求链中断
    }

    /** 过滤器销毁（无需清理资源） */
    @Override
    public void destroy() {
    }
}

     * 使用 PreparedStatement 防止 SQL注入
     *
     * @param username 用户名
     * @param password 密码
     * @return 匹配成功返回 User 对象，失败返回 null

    public int save(User user) {

            conn = DBUtil.getConnection();
            // ★ 第二个参数指定返回自增键

public class CartDao {

    /**
     * ★ 获取用户购物车 — JOIN 查询关联商品名称、品牌、价格、图片
     *
     * SQL 解析：
     *   SELECT c.*, p.name AS product_name, ...
     *   FROM cart c JOIN products p ON c.product_id = p.id
     *   WHERE c.user_id = ?
     *
     * 为什么要 JOIN？
     *   购物车表只存 product_id，展示时需要商品名称、价格等
     *   通过一次 JOIN 查询获取所有数据，避免 N+1 查询问题
     */
    public List<CartItem> findByUserId(int userId) {
        List<CartItem> list = new ArrayList<>();
        String sql = "SELECT c.*, p.name AS product_name, p.brand AS product_brand, " +
                     "p.price AS product_price, p.image AS product_image " +
                     "FROM cart c JOIN products p ON c.product_id = p.id " +
                     "WHERE c.user_id = ? ORDER BY c.create_time DESC";
        Connection conn = null;

package com.shop.dao;

import com.shop.model.Order;
import com.shop.model.OrderItem;
import com.shop.util.DBUtil;
import java.sql.*;
import java.util.ArrayList;
import java.util.List;

/**
 * 订单数据访问对象
 *
 * ★ 事务管理关键点（createOrder 方法）：
 *   1. setAutoCommit(false) 开启事务
 *   2. 插入订单主表
 *   3. 批量插入订单明细（addBatch + executeBatch）
 *   4. commit() 提交或 rollback() 回滚
 *   5. finally 中恢复 setAutoCommit(true)
 */
public class OrderDao {

    /**
     * ★ 创建订单（包含事务管理和批量插入）
     *
     * 流程：
     *   1. 开启事务
     *   2. 插入 orders 表获取 orderId
     *   3. 批量插入 order_items 表
     *   4. 提交事务
     *   5. 任何异常 → 回滚
     *
     * @return 成功返回订单ID，失败返回 -1
     */
    public int createOrder(Order order) {
        String sql = "INSERT INTO orders (user_id, order_no, total_amount, status, receiver_name, receiver_phone, receiver_address) " +
                     "VALUES (?, ?, ?, ?, ?, ?, ?)";
        Connection conn = null;
        PreparedStatement ps = null;
        try {
            conn = DBUtil.getConnection();
            conn.setAutoCommit(false);  // ★ 关闭自动提交，开启事务

            // 第1步：插入订单主表
            ps = conn.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS);
            ps.setInt(1, order.getUserId());
            ps.setString(2, order.getOrderNo());
            ps.setDouble(3, order.getTotalAmount());
            ps.setString(4, order.getStatus());
            ps.setString(5, order.getReceiverName());
            ps.setString(6, order.getReceiverPhone());
            ps.setString(7, order.getReceiverAddress());
            ps.executeUpdate();

            ResultSet rs = ps.getGeneratedKeys();
            int orderId = -1;
            if (rs.next()) {
                orderId = rs.getInt(1);  // 获取自增订单ID
            }

            // 第2步：批量插入订单明细
            String itemSql = "INSERT INTO order_items (order_id, product_id, product_name, product_price, quantity, subtotal) " +
                             "VALUES (?, ?, ?, ?, ?, ?)";
            PreparedStatement itemPs = conn.prepareStatement(itemSql);
            for (OrderItem item : order.getItems()) {
                itemPs.setInt(1, orderId);
                itemPs.setInt(2, item.getProductId());
                itemPs.setString(3, item.getProductName());    // 快照商品名
                itemPs.setDouble(4, item.getProductPrice());   // 快照单价
                itemPs.setInt(5, item.getQuantity());
                itemPs.setDouble(6, item.getSubtotal());
                itemPs.addBatch();  // ★ 加入批处理队列
            }
            itemPs.executeBatch();  // ★ 一次性执行所有INSERT

            conn.commit();  // ★ 提交事务，数据真正写入数据库
            return orderId;
        } catch (SQLException e) {
            try {
                if (conn != null) conn.rollback();  // ★ 异常回滚，保证数据一致性
            } catch (SQLException ex) { ex.printStackTrace(); }
            e.printStackTrace();
        } finally {
            try {
                if (conn != null) conn.setAutoCommit(true);  // ★ 恢复自动提交
            } catch (SQLException e) { e.printStackTrace(); }
            DBUtil.close(conn, ps);
        }
        return -1;
    }

    /**
     * ★ 获取用户的所有订单（含明细）
     *
     * 每次循环中调用 findItemsByOrderId() 加载明细
     * 注意：这会产生 N+1 查询（1次查订单 + N次查明细）
     * 对于小型项目可接受，大型项目应用 JOIN + MyBatis 等框架
     */
    public List<Order> findByUserId(int userId) {
        List<Order> list = new ArrayList<>();
        String sql = "SELECT * FROM orders WHERE user_id = ? ORDER BY create_time DESC";
        Connection conn = null;
        PreparedStatement ps = null;
        ResultSet rs = null;
        try {
            conn = DBUtil.getConnection();
            ps = conn.prepareStatement(sql);
            ps.setInt(1, userId);
            rs = ps.executeQuery();
            while (rs.next()) {
                Order order = extractOrder(rs);
                order.setItems(findItemsByOrderId(order.getId()));  // 加载明细
                list.add(order);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        } finally {
            DBUtil.close(conn, ps, rs);
        }
        return list;
    }

package com.shop.servlet;

import com.shop.dao.UserDao;
import com.shop.model.User;
import javax.servlet.ServletException;
import javax.servlet.http.*;
import java.io.IOException;

/**
 * 用户登录 Servlet
 *
 * ★ 映射路径：/user/login
 * ★ GET 请求：转发到 login.jsp 登录页面
 * ★ POST 请求：接收表单提交，验证用户名密码
 *
 * ★ Session 机制：
 *   登录成功后将 User 对象存入 req.getSession()
 *   后续请求通过 session.getAttribute("user") 获取当前登录用户
 *   退出登录时调用 session.invalidate() 销毁 Session
 */
public class LoginServlet extends HttpServlet {

    private UserDao userDao = new UserDao();

    /**
     * GET /user/login — 显示登录页面
     */
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        // forward 是服务器内部转发，URL不变，request 属性保留
        req.getRequestDispatcher("/jsp/login.jsp").forward(req, resp);
    }

    /**
     * POST /user/login — 处理登录表单
     *
     * 流程：
     *   1. 获取 username, password 参数
     *   2. 非空校验（前后端双重验证）
     *   3. 调用 UserDao 查询数据库
     *   4. 成功 → Session 存储 User → 重定向到首页
     *   5. 失败 → 设置 error 属性 → forward 回登录页显示错误
     */
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String username = req.getParameter("username");
        String password = req.getParameter("password");

        // ★ 非空校验
        if (username == null || password == null || username.trim().isEmpty() || password.trim().isEmpty()) {
            req.setAttribute("error", "用户名和密码不能为空！");
            req.getRequestDispatcher("/jsp/login.jsp").forward(req, resp);
            return;  // ★ 提前返回，不再执行后续代码
        }

        // ★ 查询用户
        User user = userDao.findByUsernameAndPassword(username.trim(), password.trim());

        if (user != null) {
            // ★ 登录成功 — 用户信息存入 Session
            req.getSession().setAttribute("user", user);
            // ★ 重定向到首页（客户端跳转，URL会变）
            //   使用 redirect 而非 forward 是为了避免刷新页面时重复提交表单
            resp.sendRedirect(req.getContextPath() + "/jsp/index.jsp");
        } else {
            req.setAttribute("error", "用户名或密码错误！");
            req.getRequestDispatcher("/jsp/login.jsp").forward(req, resp);
        }
    }
}

package com.shop.servlet;

import com.shop.dao.UserDao;
import com.shop.model.User;
import javax.servlet.ServletException;
import javax.servlet.http.*;
import java.io.IOException;

/**
 * 用户注册 Servlet
 *
 * ★ 注册验证流程：
 *   1. 用户名非空
 *   2. 密码长度 ≥ 6位
 *   3. 两次密码一致
 *   4. 用户名未被占用
 *   5. 写入数据库
 *   6. 注册成功后自动登录（直接写Session）
 */
public class RegisterServlet extends HttpServlet {

    private UserDao userDao = new UserDao();

    /** GET — 显示注册页面 */
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        req.getRequestDispatcher("/jsp/register.jsp").forward(req, resp);
    }

    /** POST — 处理注册表单 */
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String username = req.getParameter("username");
        String password = req.getParameter("password");
        String repassword = req.getParameter("repassword");
        String email = req.getParameter("email");
        String phone = req.getParameter("phone");
        String address = req.getParameter("address");

        // ★ 逐步验证，每个失败直接返回对应错误信息

        if (username == null || username.trim().isEmpty()) {
            req.setAttribute("error", "用户名不能为空！");
            req.getRequestDispatcher("/jsp/register.jsp").forward(req, resp);
            return;
        }

        if (password == null || password.trim().length() < 6) {
            req.setAttribute("error", "密码长度不能少于6位！");
            req.getRequestDispatcher("/jsp/register.jsp").forward(req, resp);
            return;
        }

        if (!password.equals(repassword)) {
            req.setAttribute("error", "两次密码输入不一致！");
            req.getRequestDispatcher("/jsp/register.jsp").forward(req, resp);
            return;
        }

        // ★ 检查用户名是否已存在
        User existUser = userDao.findByUsername(username.trim());
        if (existUser != null) {
            req.setAttribute("error", "该用户名已被注册！");
            req.getRequestDispatcher("/jsp/register.jsp").forward(req, resp);
            return;
        }

        // ★ 构建 User 对象
        User user = new User();
        user.setUsername(username.trim());
        user.setPassword(password.trim());
        user.setEmail(email != null ? email.trim() : "");
        user.setPhone(phone != null ? phone.trim() : "");
        user.setAddress(address != null ? address.trim() : "");

        int result = userDao.save(user);  // 写入数据库，返回新ID
        if (result > 0) {
            // ★ 注册成功，查询完整用户信息，自动登录
            User savedUser = userDao.findById(result);
            req.getSession().setAttribute("user", savedUser);
            resp.sendRedirect(req.getContextPath() + "/jsp/index.jsp");
        } else {
            req.setAttribute("error", "注册失败，请稍后重试！");
            req.getRequestDispatcher("/jsp/register.jsp").forward(req, resp);
        }
    }
}

package com.shop.servlet;

import com.shop.dao.ProductDao;
import com.shop.model.Product;
import javax.servlet.ServletException;
import javax.servlet.http.*;
import java.io.IOException;
import java.util.List;

/**
 * 商品 Servlet — 处理商品列表、搜索、详情、分类/品牌筛选
 *
 * ★ action 参数路由：
 *   list     → 全部商品列表
 *   detail   → 商品详情页（需传 id）
 *   search   → 搜索商品（需传 keyword）
 *   category → 按分类筛选（需传 cat）
 *   brand    → 按品牌筛选（需传 name）
 */
public class ProductServlet extends HttpServlet {

    private ProductDao productDao = new ProductDao();

    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String action = req.getParameter("action");
        if (action == null) {
            action = "list";  // 默认显示全部商品
        }

        // ★ 使用 switch 根据 action 分发到不同的处理方法
        switch (action) {
            case "detail":
                showDetail(req, resp);
                break;
            case "search":
                search(req, resp);
                break;
            case "category":
                listByCategory(req, resp);
                break;
            case "brand":
                listByBrand(req, resp);
                break;
            default:
                listAll(req, resp);
                break;
        }
    }

    /** 全部商品列表 */
    private void listAll(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        List<Product> products = productDao.findAll();
        req.setAttribute("products", products);
        req.getRequestDispatcher("/jsp/product_list.jsp").forward(req, resp);
    }

    /** 商品详情 — 通过ID查询单个商品 */
    private void showDetail(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String idStr = req.getParameter("id");
        if (idStr != null) {
            int id = Integer.parseInt(idStr);
            Product product = productDao.findById(id);
            if (product != null) {
                req.setAttribute("product", product);
                req.getRequestDispatcher("/jsp/product_detail.jsp").forward(req, resp);
                return;
            }
        }
        // 商品不存在或无ID → 重定向到列表页
        resp.sendRedirect(req.getContextPath() + "/product?action=list");
    }

    /** 模糊搜索 — 按名称/品牌/描述搜索 */
    private void search(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String keyword = req.getParameter("keyword");
        List<Product> products;
        if (keyword != null && !keyword.trim().isEmpty()) {
            products = productDao.search(keyword.trim());
            req.setAttribute("keyword", keyword.trim());  // 回显搜索关键词
        } else {
            products = productDao.findAll();  // 空关键字显示全部
        }
        req.setAttribute("products", products);
        req.getRequestDispatcher("/jsp/product_list.jsp").forward(req, resp);
    }

    /** 按分类筛选 */
    private void listByCategory(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String category = req.getParameter("cat");
        List<Product> products = productDao.findByCategory(category);
        req.setAttribute("products", products);
        req.setAttribute("currentCategory", category);  // 高亮当前分类
        req.getRequestDispatcher("/jsp/product_list.jsp").forward(req, resp);
    }

    /** 按品牌筛选 */
    private void listByBrand(HttpServletRequest req, HttpServletResponse resp)

package com.shop.servlet;

import com.shop.dao.CartDao;
import com.shop.dao.ProductDao;
import com.shop.model.CartItem;
import com.shop.model.Product;
import com.shop.model.User;
import javax.servlet.ServletException;
import javax.servlet.http.*;
import java.io.IOException;
import java.util.List;

/**
 * 购物车 Servlet
 *
 * ★ 权限检查：所有操作都需要登录
 *   未登录用户访问购物车 → 重定向到登录页
 *
 * ★ action 路由：
 *   list   → 查看购物车
 *   add    → 添加到购物车（需传 productId, quantity）
 *   update → 更新数量（需传 id, quantity）
 *   delete → 删除购物车项（需传 id）
 *
 * ★ 添加到购物车逻辑：
 *   如果该商品已在购物车 → 累加数量
 *   如果该商品不在购物车 → 新增记录
 */
public class CartServlet extends HttpServlet {

    private CartDao cartDao = new CartDao();
    private ProductDao productDao = new ProductDao();

    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        // ★ 权限检查：从 Session 获取当前登录用户
        User user = (User) req.getSession().getAttribute("user");
        if (user == null) {
            resp.sendRedirect(req.getContextPath() + "/user/login");
            return;
        }

        String action = req.getParameter("action");
        if (action == null) action = "list";

        switch (action) {
            case "add":
                addToCart(req, resp, user);
                break;
            case "update":
                updateCart(req, resp);
                break;
            case "delete":
                deleteFromCart(req, resp);
                break;
            default:
                listCart(req, resp, user);
                break;
        }
    }

    /** POST 请求委托给 doGet 处理 */
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        doGet(req, resp);
    }

    /** ★ 查看购物车 — 计算总价 */
    private void listCart(HttpServletRequest req, HttpServletResponse resp, User user)
            throws ServletException, IOException {
        List<CartItem> cartItems = cartDao.findByUserId(user.getId());
        // ★ Java 8 Stream API 计算订单总价
        double total = cartItems.stream().mapToDouble(CartItem::getSubtotal).sum();
        req.setAttribute("cartItems", cartItems);
        req.setAttribute("total", total);
        req.getRequestDispatcher("/jsp/cart.jsp").forward(req, resp);
    }

    /** ★ 添加到购物车 — 如果已存在则累加数量 */
    private void addToCart(HttpServletRequest req, HttpServletResponse resp, User user)
            throws IOException {
        String productIdStr = req.getParameter("productId");
        String quantityStr = req.getParameter("quantity");

        if (productIdStr == null) {
            resp.sendRedirect(req.getContextPath() + "/product?action=list");
            return;
        }

        int productId = Integer.parseInt(productIdStr);
        int quantity = (quantityStr != null) ? Integer.parseInt(quantityStr) : 1;

        // ★ 验证商品存在且有库存
        Product product = productDao.findById(productId);
        if (product == null || product.getStock() < 1) {
            resp.sendRedirect(req.getContextPath() + "/product?action=list");
            return;
        }

        // ★ 判断是新增还是更新
        CartItem existItem = cartDao.findByUserAndProduct(user.getId(), productId);
        if (existItem != null) {
            // 已存在 → 累加数量
            int newQuantity = existItem.getQuantity() + quantity;
            cartDao.updateQuantity(existItem.getId(), newQuantity);
        } else {
            // 不存在 → 新增记录
            CartItem item = new CartItem();
            item.setUserId(user.getId());
            item.setProductId(productId);
            item.setQuantity(quantity);
            cartDao.add(item);
        }
        resp.sendRedirect(req.getContextPath() + "/cart?action=list");
    }

    /** ★ 更新购物车数量（数量≤0时自动删除） */
    private void updateCart(HttpServletRequest req, HttpServletResponse resp)
            throws IOException {
        String idStr = req.getParameter("id");
        String quantityStr = req.getParameter("quantity");
        if (idStr != null && quantityStr != null) {
            int id = Integer.parseInt(idStr);
            int quantity = Integer.parseInt(quantityStr);
            if (quantity > 0) {
                cartDao.updateQuantity(id, quantity);
            } else {
                cartDao.delete(id);  // 数量≤0时自动移除
            }
        }
        resp.sendRedirect(req.getContextPath() + "/cart?action=list");
    }

    /** ★ 从购物车删除商品 */
    private void deleteFromCart(HttpServletRequest req, HttpServletResponse resp)
            throws IOException {
        String idStr = req.getParameter("id");
        if (idStr != null) {
            cartDao.delete(Integer.parseInt(idStr));
        }
        resp.sendRedirect(req.getContextPath() + "/cart?action=list");
    }
}

package com.shop.servlet;

import com.shop.dao.CartDao;
import com.shop.dao.OrderDao;
import com.shop.dao.ProductDao;
import com.shop.model.*;
import javax.servlet.ServletException;
import javax.servlet.http.*;
import java.io.IOException;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.List;

/**
 * 订单 Servlet
 *
 * ★ 核心流程：
 *   1. 用户点击"去结算" → showCreateOrder（显示确认页）
 *   2. 用户填写收货信息 → submitOrder（创建订单）
 *   3. 创建订单时：生成订单号 → 插入数据库（事务） → 扣库存 → 清空购物车
 */
public class OrderServlet extends HttpServlet {

    private OrderDao orderDao = new OrderDao();
    private CartDao cartDao = new CartDao();
    private ProductDao productDao = new ProductDao();

    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        // ★ 权限检查
        User user = (User) req.getSession().getAttribute("user");
        if (user == null) {
            resp.sendRedirect(req.getContextPath() + "/user/login");
            return;
        }

        String action = req.getParameter("action");
        if (action == null) action = "list";

        switch (action) {
            case "create":
                showCreateOrder(req, resp, user);  // 显示下单确认页
                break;
            default:
                listOrders(req, resp, user);       // 我的订单列表
                break;
        }
    }

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        User user = (User) req.getSession().getAttribute("user");
        if (user == null) {
            resp.sendRedirect(req.getContextPath() + "/user/login");
            return;
        }

        String action = req.getParameter("action");
        if ("submit".equals(action)) {
            submitOrder(req, resp, user);  // ★ 提交订单
        } else {
            doGet(req, resp);
        }
    }

    /** 显示下单确认页面 — 展示购物车商品和收货信息表单 */
    private void showCreateOrder(HttpServletRequest req, HttpServletResponse resp, User user)
            throws ServletException, IOException {
        List<CartItem> cartItems = cartDao.findByUserId(user.getId());
        if (cartItems.isEmpty()) {
            resp.sendRedirect(req.getContextPath() + "/cart?action=list");
            return;
        }
        double total = cartItems.stream().mapToDouble(CartItem::getSubtotal).sum();
        req.setAttribute("cartItems", cartItems);
        req.setAttribute("total", total);
        req.setAttribute("user", user);
        req.getRequestDispatcher("/jsp/order_create.jsp").forward(req, resp);
    }

    /**
     * ★ 提交订单 — 核心业务方法
     *
     * 步骤：
     *   1. 获取收货信息
     *   2. 获取购物车商品
     *   3. 生成唯一订单编号（MO + 时间戳 + 用户ID）
     *   4. 构建 Order + OrderItem 对象
     *   5. 调用 OrderDao.createOrder()（内部使用事务）
     *   6. 遍历购物车逐项扣减库存
     *   7. 清空用户购物车
     *   8. 重定向到订单列表
     */
    private void submitOrder(HttpServletRequest req, HttpServletResponse resp, User user)
            throws IOException {
        String receiverName = req.getParameter("receiverName");
        String receiverPhone = req.getParameter("receiverPhone");
        String receiverAddress = req.getParameter("receiverAddress");

        List<CartItem> cartItems = cartDao.findByUserId(user.getId());
        if (cartItems.isEmpty()) {
            resp.sendRedirect(req.getContextPath() + "/cart?action=list");
            return;
        }

        double total = cartItems.stream().mapToDouble(CartItem::getSubtotal).sum();

        // ★ 生成订单编号：MO + 年月日时分秒 + 用户ID补零4位
        String orderNo = "MO" +
            new SimpleDateFormat("yyyyMMddHHmmss").format(new Date()) +
            String.format("%04d", user.getId());

        // ★ 构建订单对象
        Order order = new Order();
        order.setUserId(user.getId());
        order.setOrderNo(orderNo);
        order.setTotalAmount(total);
        order.setStatus("待发货");  // 初始状态
        // 收货人信息默认使用用户的注册信息
        order.setReceiverName(receiverName != null ? receiverName : user.getUsername());
        order.setReceiverPhone(receiverPhone != null ? receiverPhone : (user.getPhone() != null ? user.getPhone() : ""));
        order.setReceiverAddress(receiverAddress != null ? receiverAddress : (user.getAddress() != null ? user.getAddress() : ""));

        // ★ 构建订单明细 + 扣减库存
        for (CartItem cartItem : cartItems) {
            OrderItem item = new OrderItem();
            item.setProductId(cartItem.getProductId());
            item.setProductName(cartItem.getProductName());    // 快照
            item.setProductPrice(cartItem.getProductPrice());  // 快照
            item.setQuantity(cartItem.getQuantity());
            item.setSubtotal(cartItem.getSubtotal());
            order.getItems().add(item);

            // ★ 每一步都扣减对应商品的库存
            productDao.updateStock(cartItem.getProductId(), cartItem.getQuantity());
        }

        int orderId = orderDao.createOrder(order);  // 事务写入数据库
        if (orderId > 0) {
            cartDao.clearByUserId(user.getId());  // 下单成功后清空购物车
            resp.sendRedirect(req.getContextPath() + "/order?action=list");
        } else {
            resp.getWriter().write("<script>alert('下单失败，请重试！');history.back();</script>");
        }
    }

    /** 查看我的订单列表 */
    private void listOrders(HttpServletRequest req, HttpServletResponse resp, User user)
            throws ServletException, IOException {
        List<Order> orders = orderDao.findByUserId(user.getId());
        req.setAttribute("orders", orders);
        req.getRequestDispatcher("/jsp/order_list.jsp").forward(req, resp);
    }
}

package com.shop.servlet;

import com.shop.dao.ProductDao;
import com.shop.model.Product;
import com.shop.model.User;
import javax.servlet.ServletException;
import javax.servlet.http.*;
import java.io.IOException;
import java.util.List;

/**
 * ★ 管理员商品管理 Servlet
 *
 * ★ 权限控制：每个方法开头都调用 isAdmin() 检查
 *   如果不是管理员 → 重定向到登录页
 *
 * ★ action 路由：
 *   list   → 商品管理列表
 *   add    → 显示添加商品表单
 *   save   → 保存新商品（POST）
 *   edit   → 显示编辑商品表单
 *   update → 更新商品（POST）
 *   delete → 删除商品
 */
public class AdminProductServlet extends HttpServlet {

    private ProductDao productDao = new ProductDao();

    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        if (!isAdmin(req, resp)) return;  // ★ 权限检查

        String action = req.getParameter("action");
        if (action == null) action = "list";

        switch (action) {
            case "add":
                req.getRequestDispatcher("/jsp/admin/product_form.jsp").forward(req, resp);
                break;
            case "edit":
                showEditForm(req, resp);  // 带数据的编辑表单
                break;
            case "delete":
                deleteProduct(req, resp);
                break;
            default:
                listProducts(req, resp);
                break;
        }
    }

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        if (!isAdmin(req, resp)) return;  // ★ 权限检查

        String action = req.getParameter("action");
        if ("save".equals(action)) {
            saveProduct(req, resp);     // 添加
        } else if ("update".equals(action)) {
            updateProduct(req, resp);   // 编辑
        }
    }

    /** 商品列表 */
    private void listProducts(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        List<Product> products = productDao.findAll();
        req.setAttribute("products", products);
        req.getRequestDispatcher("/jsp/admin/product_list.jsp").forward(req, resp);
    }

    /** 编辑表单 — 加载已有商品数据 */
    private void showEditForm(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {
        String idStr = req.getParameter("id");
        if (idStr != null) {
            Product product = productDao.findById(Integer.parseInt(idStr));
            req.setAttribute("product", product);  // ★ 回填到表单
        }
        req.getRequestDispatcher("/jsp/admin/product_form.jsp").forward(req, resp);
    }

    /** 保存新商品 */
    private void saveProduct(HttpServletRequest req, HttpServletResponse resp)
            throws IOException {
        Product product = new Product();
        product.setName(req.getParameter("name"));
        product.setBrand(req.getParameter("brand"));
        product.setDescription(req.getParameter("description"));
        product.setPrice(Double.parseDouble(req.getParameter("price")));
        product.setStock(Integer.parseInt(req.getParameter("stock")));
        product.setImage(req.getParameter("image"));
        product.setCategory(req.getParameter("category"));
        productDao.save(product);
        resp.sendRedirect(req.getContextPath() + "/admin/product?action=list");
    }

    /** 更新商品 */
    private void updateProduct(HttpServletRequest req, HttpServletResponse resp)
            throws IOException {
        Product product = new Product();
        product.setId(Integer.parseInt(req.getParameter("id")));  // ★ 必须传ID
        product.setName(req.getParameter("name"));
        product.setBrand(req.getParameter("brand"));
        product.setDescription(req.getParameter("description"));
        product.setPrice(Double.parseDouble(req.getParameter("price")));
        product.setStock(Integer.parseInt(req.getParameter("stock")));
        product.setImage(req.getParameter("image"));
        product.setCategory(req.getParameter("category"));
        productDao.update(product);
        resp.sendRedirect(req.getContextPath() + "/admin/product?action=list");
    }

    /** 删除商品 */
    private void deleteProduct(HttpServletRequest req, HttpServletResponse resp)
            throws IOException {
        String idStr = req.getParameter("id");
        if (idStr != null) {
            productDao.delete(Integer.parseInt(idStr));
        }
        resp.sendRedirect(req.getContextPath() + "/admin/product?action=list");
    }

    /**
     * ★ 检查管理员权限
     * 从 Session 获取 user，验证 role == "admin"
     * @return true=管理员, false=已重定向到登录页
     */
    private boolean isAdmin(HttpServletRequest req, HttpServletResponse resp) throws IOException {
        User user = (User) req.getSession().getAttribute("user");
        if (user == null || !"admin".equals(user.getRole())) {
            resp.sendRedirect(req.getContextPath() + "/user/login");
            return false;
        }
        return true;
    }
}

十、系统架构总结
================================================================================

1. ★ 请求处理流程 ★
   浏览器 → EncodingFilter(设UTF-8) → 目标Servlet → DAO(操作DB) → JSP(渲染) → 浏览器

2. ★ 数据流转 ★
   Servlet 处理业务逻辑，把数据通过 req.setAttribute() 传给 JSP
   JSP 通过 EL表达式 ${xxx} 和 JSTL标签 <c:xxx> 展示数据

3. ★ 权限控制 ★
   - 普通用户：Session 中 user != null
   - 管理员：user.role == "admin"
   - 商品浏览不强制登录
   - 购物车、下单需要登录

4. ★ 订单状态流转 ★
   待发货 → (管理员发货) → 已发货 → (管理员完成) → 已完成

5. ★ 在页面上修复的问题 ★
   - JSP EL 中不能用 String.format() → 改用 <fmt:formatNumber>
   - JSP EL 中 + 不能拼接字符串 → 拆分 <c:if> 分支
   - 图片 onerror 无限循环 → 加 this.onerror=null + 创建 default.svg
   - 数据库中文默认值报错 → 关闭 sql_mode 严格模式后建表

6. ★ 不足之处 ★
   - 密码明文存储（生产环境应使用 BCrypt 加密）
   - 数据库直连无连接池（生产环境应使用 HikariCP/Druid）
   - N+1 查询问题（订单列表逐条查明细）
   - 无分页功能
   - 无输入校验和XSS防护
